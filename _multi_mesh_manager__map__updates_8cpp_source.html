39std::tuple<dart::Dart, dart::Dart>
40MultiMeshManager::mapped_darts(const Mesh& my_mesh, const Mesh& child_mesh, int64_t index) const
41{
42 assert(&my_mesh.m_multi_mesh_manager == this);
43 auto [parent_to_child_accessor, child_to_parent_accessor] =
44 get_map_const_accessors(my_mesh, child_mesh);
45 const PrimitiveType parent_pt = my_mesh.top_simplex_type();
46 const PrimitiveType child_pt = child_mesh.top_simplex_type();
47 wmtk::dart::Dart parent_to_child_dart =
48 parent_to_child_accessor.IndexBaseType::operator[](index);
49 if (parent_to_child_dart.is_null()) {
50 return {};
51 }
52 // the child to parent is always the global id
53 wmtk::dart::Dart child_to_parent_dart =
54 child_to_parent_accessor.IndexBaseType::operator[](parent_to_child_dart.global_id());
55
56 const dart::SimplexDart& child_sd = dart::SimplexDart::get_singleton(child_pt);
57 int8_t child_permutation = child_sd.identity();
58 int8_t parent_permutation = parent_to_child_dart.permutation();
59 if (parent_pt == child_pt) {
60 std::swap(child_permutation, parent_permutation);
61 }
62
63 dart::Dart parent_dart = dart::Dart(child_to_parent_dart.global_id(), parent_permutation);
64 dart::Dart child_dart = dart::Dart(parent_to_child_dart.global_id(), child_permutation);
65 return {parent_dart, child_dart};
66
67#if !defined(NDEBUG)
68 const dart::SimplexDart& parent_sd = dart::SimplexDart::get_singleton(parent_pt);
69 assert(
70 wmtk::dart::utils::apply_simplex_involution(
71 parent_pt,
72 child_pt,
73 parent_to_child_dart,
74 parent_dart) == child_dart);
75
76#endif
77}

102 return wmtk::multimesh::utils::vectors_to_tuples(parent_to_child_data);

145 // are evaluated by a multimesh topological guarantee or handled by operation specific

147 // an operation

149 if (children().empty()) {
150 return;

157 const std::vector<int64_t>& gids = operation_data.global_ids_to_update[get_primitive_type_id(
158 primitive_type)]; // get facet gids(primitive_type);
159
160 // go over every child mesh and try to update their hashes
161 for (auto& child_data : children()) {
162 auto& child_mesh = *child_data.mesh;
163 // ignore ones whos map are the wrong dimension
164 if (child_mesh.top_simplex_type() != primitive_type) {

167 // logger().trace(

175 // auto child_flag_accessor = child_mesh.get_const_flag_accessor(primitive_type);

219 "Updating tuple gids {} {}",

221 std::string(child_tuple));
222 wmtk::multimesh::utils::symmetric_write_tuple_map_attributes(

514 const PrimitiveType primitive_type = old_simplex.primitive_type();
515
516 for (const auto& [old_cid, new_cids] : split_cell_maps) {

518 continue;
519 }
520
521 auto old_tuple_opt =
522 find_tuple_from_gid(my_mesh, my_mesh.top_simplex_type(), tuple_alternatives, old_cid);
523 // assert(old_tuple_opt.has_value());
524 if (!(old_tuple_opt.has_value())) {
525 return std::optional<Tuple>{};
526 }
527
528 const Tuple& old_cid_tuple = old_tuple_opt.value();
529 for (const int64_t new_cid : new_cids) {
530 Tuple tuple(
531 old_cid_tuple.local_vid(),
532 old_cid_tuple.local_eid(),
533 old_cid_tuple.local_fid(),
534 new_cid);
535
536
537 if (my_mesh.is_valid(tuple) && !my_mesh.is_removed(tuple) &&
538 old_simplex_gid == my_mesh.id(tuple, primitive_type)) {
539 return tuple;
540 }
541 }
542 }
543 return std::optional<Tuple>{};
544}
