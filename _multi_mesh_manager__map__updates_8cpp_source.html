38std::tuple<dart::Dart, dart::Dart>
39MultiMeshManager::mapped_darts(const Mesh& my_mesh, const Mesh& child_mesh, int64_t index) const
40{
41 assert(&my_mesh.m_multi_mesh_manager == this);
42 auto [parent_to_child_accessor, child_to_parent_accessor] =
43 get_map_const_accessors(my_mesh, child_mesh);
44 const PrimitiveType parent_pt = my_mesh.top_simplex_type();
45 const PrimitiveType child_pt = child_mesh.top_simplex_type();
46 wmtk::dart::Dart parent_to_child_dart =
47 parent_to_child_accessor.IndexBaseType::operator[](index);
48 if (parent_to_child_dart.is_null()) {
49 return {};
50 }
51 // the child to parent is always the global id
52 wmtk::dart::Dart child_to_parent_dart =
53 child_to_parent_accessor.IndexBaseType::operator[](parent_to_child_dart.global_id());
54
55 const dart::SimplexDart& child_sd = dart::SimplexDart::get_singleton(child_pt);
56 int8_t child_permutation = child_sd.identity();
57 int8_t parent_permutation = parent_to_child_dart.permutation();
58 if (parent_pt == child_pt) {
59 std::swap(child_permutation, parent_permutation);
60 }
61
62 dart::Dart parent_dart = dart::Dart(child_to_parent_dart.global_id(), parent_permutation);
63 dart::Dart child_dart = dart::Dart(parent_to_child_dart.global_id(), child_permutation);
64 return {parent_dart, child_dart};
65
66#if !defined(NDEBUG)
67 const dart::SimplexDart& parent_sd = dart::SimplexDart::get_singleton(parent_pt);
68 assert(
69 wmtk::dart::utils::apply_simplex_involution(
70 parent_pt,
71 child_pt,
72 parent_to_child_dart,
73 parent_dart) == child_dart);
74
75#endif
76}

78std::tuple<Tuple, Tuple>
79MultiMeshManager::mapped_tuples(const Mesh& my_mesh, const Mesh& child_mesh, int64_t index) const
80{
81 auto [parent_dart, child_dart] = mapped_darts(my_mesh, child_mesh, index);
82
83 const PrimitiveType parent_pt = my_mesh.top_simplex_type();
84 const PrimitiveType child_pt = child_mesh.top_simplex_type();
85 const dart::SimplexDart& parent_sd = dart::SimplexDart::get_singleton(parent_pt);
86 const dart::SimplexDart& child_sd = dart::SimplexDart::get_singleton(child_pt);
87 Tuple parent_tuple = parent_sd.tuple_from_dart(parent_dart);
88 Tuple child_tuple = child_sd.tuple_from_dart(child_dart);
89
90 return {parent_tuple, child_tuple};

112 parent_to_child_map_attribute_name(i));

127void MultiMeshManager::update_maps_from_edge_operation(
128 Mesh& my_mesh,
129 PrimitiveType primitive_type,
130 const operations::EdgeOperationData& operation_data)
131{
132 // Assumes that the child mesh does not have a simplex affected by the operation (such cases
133 // are evaluated by a multimesh topological guarantee or handled by operation specific
134 // code. This just makes sure that the tuple used to map to a child mesh still exists after
135 // an operation
136
137 if (children().empty()) {
138 return;
139 }
140 // auto parent_flag_accessor = my_mesh.get_const_flag_accessor(primitive_type);
141 // auto& update_tuple = [&](const auto& flag_accessor, Tuple& t) -> bool {
142 // if(acc.index_access().
143 // };
144
145 const std::vector<int64_t>& gids = operation_data.global_ids_to_update[get_primitive_type_id(
146 primitive_type)]; // get facet gids(primitive_type);
147 // spdlog::info("Updating map for {}-simplices from edge split on [{}]:{}, ids are {}",
148 // primitive_type,my_mesh.absolute_multi_mesh_id(), operation_data.m_input_edge_gid, gids);
149
150 // go over every child mesh and try to update their hashes
151 for (auto& child_data : children()) {
152 auto& child_mesh = *child_data.mesh;
153 // ignore ones whos map are the wrong dimension
154 if (child_mesh.top_simplex_type() != primitive_type) {
155 continue;
156 }
157 // spdlog::info(
158 // "[{}->{}] Doing a child mesh ({})",
159 // fmt::join(my_mesh.absolute_multi_mesh_id(), ","),
160 // fmt::join(child_mesh.absolute_multi_mesh_id(), ","),child_mesh.top_simplex_type());
161 // get accessors to the maps
162 auto maps = get_map_accessors(my_mesh, child_data);
163 auto& [parent_to_child_accessor, child_to_parent_accessor] = maps;
164
165 // auto child_flag_accessor = child_mesh.get_const_flag_accessor(primitive_type);
166
167
168 spdlog::info("Checking for updating {}-simplices with gids {}", primitive_type, fmt::join(gids,","));
169 for (const auto& gid : gids) {
170 const bool parent_exists = !my_mesh.is_removed(gid, primitive_type);
171 if (!parent_exists) {
172 continue;
173 }
174
175 auto [parent_tuple, child_tuple] = mapped_tuples(my_mesh, *child_data.mesh, gid);
176
177
178 // If the parent tuple is valid, it means this parent-child pair has already been
179 // handled, so we can skip it
180 // If the parent tuple is invalid then there was no map so we can try the next cell
181 if (parent_tuple.is_null()) {
182 continue;
183 }
184
185
186 // it is not this function's responsibility to handle cases where
187 // teh child mesh's topology has changed on the mapped simplex, so we must make sure
188 // it's still there
189 const bool child_exists = !child_mesh.is_removed(child_tuple);
190 if (!child_exists) {
191 continue;
192 }
193
194 Tuple old_parent_tuple = parent_tuple;
195
196 parent_tuple = wmtk::multimesh::find_valid_tuple(
197 my_mesh,
198 parent_tuple,
199 primitive_type,
200 operation_data);
201 spdlog::warn("Parent {} moved to {} to map to child {}", std::string(old_parent_tuple), std::string(parent_tuple), std::string(child_tuple));
202
203 if (parent_tuple.is_null()) {
204 continue;
205 }
206 assert(my_mesh.is_valid(parent_tuple));
207
208 assert(!parent_tuple.is_null());
209 wmtk::multimesh::utils::symmetric_write_tuple_map_attributes(
210 parent_to_child_accessor,
211 child_to_parent_accessor,
212 parent_tuple,
213 child_tuple);
214 }
215 }
