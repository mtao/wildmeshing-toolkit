38std::tuple<dart::Dart, dart::Dart>
39MultiMeshManager::mapped_darts(const Mesh& my_mesh, const Mesh& child_mesh, int64_t index) const
40{
41 assert(&my_mesh.m_multi_mesh_manager == this);
42 auto [parent_to_child_accessor, child_to_parent_accessor] =
43 get_map_const_accessors(my_mesh, child_mesh);
44 const PrimitiveType parent_pt = my_mesh.top_simplex_type();
45 const PrimitiveType child_pt = child_mesh.top_simplex_type();
46 wmtk::dart::Dart parent_to_child_dart =
47 parent_to_child_accessor.IndexBaseType::operator[](index);
48 if (parent_to_child_dart.is_null()) {
49 return {};
50 }
51 assert(!child_mesh.is_removed(parent_to_child_dart.global_id()));
52 // the child to parent is always the global id
53 wmtk::dart::Dart child_to_parent_dart =
54 child_to_parent_accessor.IndexBaseType::operator[](parent_to_child_dart.global_id());
55
56 const dart::SimplexDart& child_sd = dart::SimplexDart::get_singleton(child_pt);
57 int8_t child_permutation = child_sd.identity();
58 int8_t parent_permutation = parent_to_child_dart.permutation();
59 if (parent_pt == child_pt) {
60 std::swap(child_permutation, parent_permutation);
61 }
62
63 dart::Dart parent_dart = dart::Dart(child_to_parent_dart.global_id(), parent_permutation);
64 dart::Dart child_dart = dart::Dart(parent_to_child_dart.global_id(), child_permutation);
65
66#if !defined(NDEBUG)
67 const dart::SimplexDart& parent_sd = dart::SimplexDart::get_singleton(parent_pt);
68 assert(
69 wmtk::dart::utils::apply_simplex_involution(
70 parent_pt,
71 child_pt,
72 parent_to_child_dart,
73 parent_dart) == child_dart);
74
75#endif
76 return {parent_dart, child_dart};
77}

79std::tuple<Tuple, Tuple>
80MultiMeshManager::mapped_tuples(const Mesh& my_mesh, const Mesh& child_mesh, int64_t index) const
81{
82 auto [parent_dart, child_dart] = mapped_darts(my_mesh, child_mesh, index);
83
84 const PrimitiveType parent_pt = my_mesh.top_simplex_type();
85 const PrimitiveType child_pt = child_mesh.top_simplex_type();
86 const dart::SimplexDart& parent_sd = dart::SimplexDart::get_singleton(parent_pt);
87 const dart::SimplexDart& child_sd = dart::SimplexDart::get_singleton(child_pt);
88 Tuple parent_tuple = parent_sd.tuple_from_dart(parent_dart);
89 Tuple child_tuple = child_sd.tuple_from_dart(child_dart);
90
91 return {parent_tuple, child_tuple};
92}

112 parent_to_child_map_attribute_name(i));

127void MultiMeshManager::update_maps_from_edge_operation(
128 Mesh& my_mesh,
129 PrimitiveType primitive_type,
130 const operations::EdgeOperationData& operation_data)
131{
132 // Assumes that the child mesh does not have a simplex affected by the operation (such cases
133 // are evaluated by a multimesh topological guarantee or handled by operation specific
134 // code. This just makes sure that the tuple used to map to a child mesh still exists after
135 // an operation
136
137 if (children().empty()) {
138 return;
139 }
140 const dart::SimplexDart& parent_sd =
141 dart::SimplexDart::get_singleton(my_mesh.top_simplex_type());
142 // auto parent_flag_accessor = my_mesh.get_const_flag_accessor(primitive_type);
143 // auto& update_tuple = [&](const auto& flag_accessor, Tuple& t) -> bool {
144 // if(acc.index_access().
145 // };
146
147 const std::vector<int64_t>& gids = operation_data.global_ids_to_update[get_primitive_type_id(
148 primitive_type)]; // get facet gids(primitive_type);
149
150 // go over every child mesh and try to update their hashes
151 for (auto& child_data : children()) {
152 auto& child_mesh = *child_data.mesh;
153 // ignore ones whos map are the wrong dimension
154 if (child_mesh.top_simplex_type() != primitive_type) {
155 continue;
156 }
157 auto maps = get_map_accessors(my_mesh, child_data);
158 auto& [parent_to_child_accessor, child_to_parent_accessor] = maps;
159
160 // auto child_flag_accessor = child_mesh.get_const_flag_accessor(primitive_type);
161
162 for (const auto& gid : gids) {
163 const bool parent_exists = !my_mesh.is_removed(gid, primitive_type);
164 if (!parent_exists) {
165 continue;
166 }
167
168 auto [parent_tuple, child_tuple] = mapped_tuples(my_mesh, *child_data.mesh, gid);
169
170
171 // If the parent tuple is valid, it means this parent-child pair has already been
172 // handled, so we can skip it
173 // If the parent tuple is invalid then there was no map so we can try the next cell
174 if (parent_tuple.is_null()) {
175 continue;
176 }
177
178
179 // it is not this function's responsibility to handle cases where
180 // teh child mesh's topology has changed on the mapped simplex, so we must make sure
181 // it's still there
182 const bool child_exists = !child_mesh.is_removed(child_tuple);
183 if (!child_exists) {
184 continue;
185 }
186
187
188 parent_tuple = wmtk::multimesh::find_valid_tuple(
189 my_mesh,
190 parent_tuple,
191 primitive_type,
192 operation_data);
193 if (parent_tuple.is_null()) {
194 continue;
195 }
196 assert(my_mesh.is_valid(parent_tuple));
197
198 assert(!parent_tuple.is_null());
199 wmtk::multimesh::utils::symmetric_write_tuple_map_attributes(
200 parent_to_child_accessor,
201 child_to_parent_accessor,
202 parent_tuple,
203 child_tuple);
204 }
205 }
