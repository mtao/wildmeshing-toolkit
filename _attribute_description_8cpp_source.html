98AttributeDescription::operator std::string() const
99{
100 auto get_tok = [](char c, const auto& opt) -> std::string {
101 if (opt.has_value()) {
102 using T = typename std::decay_t<decltype(opt)>::value_type;
103 if constexpr (std::is_same_v<T, attribute::AttributeType>) {
104 if (opt.has_value()) {
105 return fmt::format("{}={}", c, attribute::attribute_type_name(opt.value()));
106 }
107 } else if constexpr (std::is_same_v<T, char>) {
108 return fmt::format("{}={:d}", c, opt.value());
109 } else {
110 return fmt::format("{}={}", c, opt.value());
111 }
112 }
113 return fmt::format("{}=?", c);
114 };
115 return fmt::format(
116 "AttributeDescription({},{},{},{})",
117 path,
118 get_tok('t', type),
119 get_tok('s', simplex_dimension),
120 get_tok('d', dimension));
121}
