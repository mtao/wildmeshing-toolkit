37 {"output_mesh_name", "tagged_edges"},
38 {"manifold_decomposition", false},
39 {"tag_attribute",
40 {
41 {"path", "/tag"},

43 {"type", "char"},

45 {"value", 1}};
46

55 auto tag_acc = mptr->create_accessor(tag_attr);
56 for (const auto& t : mptr->get_all(wmtk::PrimitiveType::Edge)) {
57 if (mptr->is_boundary_edge(t)) {
58 tag_acc.scalar_attribute(t) = 1;
59 }
60 }
61 opt.run(mc);
62 const auto& nmm = mc.get_named_multimesh("root");
63
64 auto& ring = mc.get_mesh("root.tagged_edges");
65 CHECK(ring.get_all(wmtk::PrimitiveType::Vertex).size() == 10);
66 CHECK(ring.get_all(wmtk::PrimitiveType::Edge).size() == 10);
67 REQUIRE(ring.absolute_multi_mesh_id() == std::vector<int64_t>{0});
68 REQUIRE(&ring.get_multi_mesh_root() == mptr.get());
69 for (const auto& t : ring.get_all(wmtk::PrimitiveType::Edge)) {
70 // every edge in this ring should be interior
71 CHECK(!ring.is_boundary(wmtk::PrimitiveType::Edge, t));
72 auto ps = ring.map_to_parent(wmtk::simplex::Simplex::edge(t));
73 REQUIRE(mptr->is_valid(ps));
74 CHECK(mptr->is_boundary(ps));
75 }
76 {
77 wmtk::components::multimesh::MultimeshOptions gen_opts;
78 gen_opts = tag_js;
79 auto ptr = std::dynamic_pointer_cast<wmtk::components::multimesh::MultimeshTagOptions>(
80 gen_opts.options);
81 REQUIRE(bool(ptr));
82 CHECK((*ptr) == opt);
83 }
