268void Operation::reserve_enough_simplices()
269{
270 // by default assume we'll at most create N * capacity
271 constexpr static int64_t default_preallocation_size = 3;
272
273 auto run = [&](auto&& m) {
274 if constexpr (!std::is_const_v<std::remove_reference_t<decltype(m)>>) {
275 auto cap = m.m_attribute_manager.m_capacities;
276 for (auto& v : cap) {
277 v *= default_preallocation_size;
278 }
279 m.reserve_more_attributes(cap);
280 }
281 };
282 multimesh::MultiMeshVisitor visitor(run);
283 visitor.execute_from_root(mesh());
284}
