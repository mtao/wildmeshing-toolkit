276void Operation::reserve_enough_simplices()
277{
278 // by default assume we'll at most create N * capacity
279 constexpr static int64_t default_preallocation_size = 3;
280
281 auto run = [&](auto&& m) {
282 if constexpr (!std::is_const_v<std::remove_reference_t<decltype(m)>>) {
283 auto cap = m.m_attribute_manager.m_capacities;
284 for (auto& v : cap) {
285 v *= default_preallocation_size;
286 }
287 m.reserve_more_attributes(cap);
288 }
289 };
290 multimesh::MultiMeshVisitor visitor(run);
291 visitor.execute_from_root(mesh());
292}
