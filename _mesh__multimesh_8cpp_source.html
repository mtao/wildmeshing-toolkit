134 const simplex::Simplex& my_simplex) const
135{
136 assert(is_valid(my_simplex));
137 std::map<const Mesh*, std::vector<simplex::Simplex>> ret;
138 for (const auto& child : m_multi_mesh_manager.m_children) {
139 const auto& mesh = *child.mesh;
140 auto child_simplices = map_to_child(mesh, my_simplex);
141 if (child_simplices.empty()) {
142 continue;
143 }
144
145 for (const auto& s : child_simplices) {
146 auto children = mesh.map_all_children(s);
147 ret.merge(std::move(children));
148 }
149 ret[&mesh] = std::move(child_simplices);
150 }
