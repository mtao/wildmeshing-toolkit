126 const simplex::Simplex& my_simplex) const
127{
128 assert(is_valid(my_simplex));
129 std::map<const Mesh*, std::vector<simplex::Simplex>> ret;
130 for (const auto& child : m_multi_mesh_manager.m_children) {
131 const auto& mesh = *child.mesh;
132 auto child_simplices = map_to_child(mesh, my_simplex);
133 if (child_simplices.empty()) {
134 continue;
135 }
136
137 for (const auto& s : child_simplices) {
138 auto children = mesh.map_all_children(s);
139 ret.merge(std::move(children));
