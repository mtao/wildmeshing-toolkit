92template <int8_t SimplexDim, int8_t FaceDim>
93std::map<int64_t, std::set<dart::Dart>> cofacets(const IndexSimplexMapper& ism)
94{
95 std::map<int64_t, std::set<dart::Dart>> coboundary;
96 auto S = ism.template simplices<SimplexDim>();
97 for (int j = 0; j < S.size(); ++j) {
98 const auto& s = S.at(j);
99 static_assert(std::tuple_size_v<std::decay_t<decltype(s)>> == SimplexDim + 1);
100 std::array<int64_t, FaceDim + 1> r;
101 std::copy(s.begin(), s.begin() + FaceDim, r.begin());
102
103 std::array<int64_t, SimplexDim + 1> ss = s;
104 std::array<int8_t, SimplexDim + 1> local_permutation;
105 std::iota(local_permutation.begin(), local_permutation.end(), 0);
106
107 auto add = [&]() {
108 int64_t index = ism.get_index(r);
109
110 std::array<int64_t, FaceDim + 1> sorted = r;
111 std::sort(sorted.begin(), sorted.end());
112 int8_t p = wmtk::dart::utils::get_canonical_subdart_permutation(ss, sorted);
113 auto myd = wmtk::dart::Dart(j, p);
114 coboundary[index].emplace(myd);
115 };
116
117 for (const auto& comb : combinations<SimplexDim + 1, FaceDim + 1>::values) {
118 for (size_t k = 0; k < FaceDim + 1; ++k) {
119 r[k] = s[comb[k]];
120 }
121 add();
122 }
123 }
124 return coboundary;
125}
