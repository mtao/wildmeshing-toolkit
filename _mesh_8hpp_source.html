1024template <std::ranges::forward_range ContainerType>
1025#else
1026template <typename ContainerType>
1027#endif
1028inline Tuple Mesh::switch_tuples(const Tuple& tuple, const ContainerType& sequence) const
1029{
1030 static_assert(std::is_same_v<typename ContainerType::value_type, PrimitiveType>);
1031 Tuple r = tuple;
1032 const PrimitiveType top_type = top_simplex_type();
1033
1034 const int64_t boundary_dim = top_cell_dimension() - 1;
1035 const PrimitiveType boundary_pt = static_cast<PrimitiveType>(boundary_dim);
1036
1037 for (const PrimitiveType primitive : sequence) {
1038 // for top level simplices we cannot navigate across boundaries
1039 if (primitive == top_type && is_boundary(boundary_pt, r)) {
1040 assert(!is_boundary(boundary_pt, r));
1041 r = {};
1042 return r;
1043 }
1044 r = switch_tuple(r, primitive);
1045 }
1046 return r;
1047}
