1009template <std::ranges::forward_range ContainerType>
1010#else
1011template <typename ContainerType>
1012#endif
1013inline Tuple Mesh::switch_tuples(const Tuple& tuple, const ContainerType& sequence) const
1014{
1015 static_assert(std::is_same_v<typename ContainerType::value_type, PrimitiveType>);
1016 Tuple r = tuple;
1017 const PrimitiveType top_type = top_simplex_type();
1018
1019 const int64_t boundary_dim = top_cell_dimension() - 1;
1020 const PrimitiveType boundary_pt = static_cast<PrimitiveType>(boundary_dim);
1021
1022 for (const PrimitiveType primitive : sequence) {
1023 // for top level simplices we cannot navigate across boundaries
1024 if (primitive == top_type && is_boundary(boundary_pt, r)) {
1025 assert(!is_boundary(boundary_pt, r));
1026 r = {};
1027 return r;
1028 }
1029 r = switch_tuple(r, primitive);
1030 }
1031 return r;
1032}
