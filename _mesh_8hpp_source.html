1010#endif
1011inline Tuple Mesh::switch_tuples(const Tuple& tuple, const ContainerType& sequence) const
1012{
1013 static_assert(std::is_same_v<typename ContainerType::value_type, PrimitiveType>);
1014 Tuple r = tuple;
1015 const PrimitiveType top_type = top_simplex_type();
1016
1017 const int64_t boundary_dim = top_cell_dimension() - 1;
1018 const PrimitiveType boundary_pt = static_cast<PrimitiveType>(boundary_dim);
1019
1020 for (const PrimitiveType primitive : sequence) {
1021 // for top level simplices we cannot navigate across boundaries
1022 if (primitive == top_type && is_boundary(boundary_pt, r)) {
1023 assert(!is_boundary(boundary_pt, r));
1024 r = {};
1025 return r;
1026 }
1027 r = switch_tuple(r, primitive);
1028 }
1029 return r;
1030}
