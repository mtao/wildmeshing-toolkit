106 }
107

111 configure_smooth();
112 assert(bool(m_smooth));
113 m_operations.emplace("smooth", m_smooth);

115 wmtk::logger().info("Running Isotropic Remeshing without a smooth configured");

118 if (m_options.passes.empty() && m_options.mesh_collection != nullptr) {
119 m_options.passes.emplace_back(Pass{
120 m_options.mesh_collection->get_mesh_path(m_options.position_attribute.mesh()),
121 1,
122 {"split", "collapse", "swap", "smooth"}});
123 }
124 for (size_t j = 0; j < m_options.passes.size(); ++j) {
125 auto& pass = m_options.passes[j];
126 if (pass.operations.empty()) {
127 wmtk::logger().info("Pass of isotropic remeshing didn't specify any operations, "
128 "assuming it's a split,collapse,swap,smooth");

133 throw std::runtime_error(

163void IsotropicRemeshing::make_interior_invariants()
164{
165 auto position = m_options.position_attribute;
166 Mesh& mesh = position.mesh();
167 auto invariant_interior_vertex = std::make_shared<wmtk::invariants::InvariantCollection>(mesh);
168 m_interior_edge_invariants = std::make_shared<wmtk::invariants::InvariantCollection>(mesh);
169
170 auto set_all_invariants = [&](auto&& m) {
171 // TODO: this used to do vertex+edge, but just checkign for vertex should be sufficient?
172 for (PrimitiveType pt = PrimitiveType::Vertex; pt < m.top_simplex_type(); pt = pt + 1) {
173 invariant_interior_vertex->add(
174 std::make_shared<wmtk::invariants::InteriorSimplexInvariant>(m, pt));
175 }
176
177 m_interior_edge_invariants->add(
178 std::make_shared<wmtk::invariants::InteriorSimplexInvariant>(m, PrimitiveType::Edge));
179 };
180 wmtk::multimesh::MultiMeshVisitor visitor(set_all_invariants);
181 visitor.execute_from_root(mesh);
182 m_interior_position_invariants = invariant_interior_vertex;
183}

186{

211 wmtk::components::output::output(m_options.mesh_collection->get_mesh(name), opt);
212 }
213 };
214
215 log_mesh(0);
216 size_t index = 0;
217 for (size_t k = 1; k <= m_options.iterations; ++k) {
218 for (size_t j = 0; j < m_options.passes.size(); ++j) {
219 Pass& p = m_options.passes[j];
220 wmtk::logger().info(
221 "Running pass {}/{} of iteration {}/{}. Has {} sub-iterations on {})",
222 j,
223 m_options.passes.size(),
224 k,
225 m_options.iterations,
226 p.iterations,
227 p.mesh_path);
228 run(p, j);
229 }
230 log_mesh(k);

332void IsotropicRemeshing::make_envelopes()
333{
334 if (!m_options.envelope_size.has_value()) {
335 throw std::runtime_error(
336 "Could not create envelopes because options do not have an envelope size");
337 }
338 auto envelope_positions = all_envelope_positions();
339
340 std::vector<std::shared_ptr<wmtk::invariants::EnvelopeInvariant>> envelope_invariants;
341
342
343 std::transform(
344 envelope_positions.begin(),
345 envelope_positions.end(),
346 std::back_inserter(envelope_invariants),
347 [&](const wmtk::attribute::MeshAttributeHandle& mah) {
348 spdlog::warn(
349 "Envelope made for {}",
350 primitive_type_name(mah.mesh().top_simplex_type()));
351 return std::make_shared<wmtk::invariants::EnvelopeInvariant>(
352 mah,
353 std::sqrt(2) * m_options.envelope_size.value(),
354 mah);
355 });
356
357 m_envelope_invariants = std::make_shared<wmtk::invariants::InvariantCollection>(
358 m_options.position_attribute.mesh().get_multi_mesh_root());
359
360 for (const auto& invar : envelope_invariants) {
361 m_envelope_invariants->add(invar);
362 }
363}
