51void IsotropicRemeshing::load_shared_invariants(const IsotropicRemeshingOptions& opts)
52{
53 auto position_attr = get_attribute(opts.position_attribute);
54
55 if (!position_attr.is_valid()) {
56 throw std::runtime_error("Isotropic remeshing run without a valid position attribute");
57 }
58 m_universal_invariants =
59 std::make_shared<wmtk::invariants::InvariantCollection>(position_attr.mesh());
60 for (const auto& attr : opts.improvement_attributes) {
61 m_universal_invariants->add(
62 std::make_shared<invariants::ImprovementInvariant>(get_attribute(attr)));
63 }
64
65
66 if (opts.envelope_size.has_value()) {
67 make_envelopes(opts);
68 }
69
70 make_interior_invariants(opts);

72void IsotropicRemeshing::load_transfers(const IsotropicRemeshingOptions& opts)
73{
74 for (const auto& [child, parent] : opts.copied_attributes) {
75 auto parent_attr = get_attribute(parent);
76 try {
77 auto child_attr = get_attribute(child);
78 m_operation_transfers.emplace_back(
79 wmtk::operations::attribute_update::make_cast_attribute_transfer_strategy(
80 parent_attr,
81 child_attr));
82 } catch (const components::multimesh::utils::detail::attribute_missing_error& e) {
83 // find the parent attribute in the mesh collection to flush out its full description
84 multimesh::utils::AttributeDescription child2(child.path, parent_attr);
85
86 assert(child.compatible(child2));
87
88 auto child_attr = multimesh::utils::create_attribute(m_meshes, child2);
89 m_operation_transfers.emplace_back(
90 wmtk::operations::attribute_update::make_cast_attribute_transfer_strategy(
91 parent_attr,
92 child_attr));
93 }
94 }
95 for (const auto& transfer : opts.utility_attributes) {
96 m_operation_transfers.emplace_back(transfer->create(m_meshes));
97 }
98}

101IsotropicRemeshing::IsotropicRemeshing(
102 multimesh::MeshCollection& mc,
103 const IsotropicRemeshingOptions& opts)
104 : m_meshes(mc)
105{
106 passes = opts.passes;
107 iterations = opts.iterations;
108 start_with_collapse = opts.start_with_collapse;
109
110 intermediate_output_format = opts.intermediate_output_format;
111
112
113 load_shared_invariants(opts);
114 load_transfers(opts);
115
116
117 // split
118 if (opts.split.enabled) {
119 configure_split(opts);
120 assert(bool(m_split));
121 m_operations.emplace("split", m_split);
122 } else {
123 wmtk::logger().info("Running Isotropic Remeshing without a split configured");
124 }
125
126
128 // collapse
129
130 if (opts.collapse.enabled) {
131 configure_collapse(opts);
132 assert(bool(m_collapse));
133 m_operations.emplace("collapse", m_collapse);
134 } else {
135 wmtk::logger().info("Running Isotropic Remeshing without a collapse configured");
136 }
137
138
140 // swap
141
142 if (opts.swap.enabled) {
143 configure_swap(opts);
144 assert(bool(m_swap));
145 m_operations.emplace("swap", m_swap);
146 } else if (opts.swap.mode != EdgeSwapMode::Skip) {
147 wmtk::logger().info(
148 "Running Isotropic Remeshing without a swap configured despite being "
149 "supposed to use them");
150 }
151
153 // smooth
154 if (opts.smooth.enabled) {
155 configure_smooth(opts);
156 assert(bool(m_smooth));
157 m_operations.emplace("smooth", m_smooth);
158 } else {
159 wmtk::logger().info("Running Isotropic Remeshing without a smooth configured");
160 }
161
162 if (opts.passes.empty()) {
163 auto position_attr = get_attribute(opts.position_attribute);
164
165 if (!position_attr.is_valid()) {
166 throw std::runtime_error("Isotropic remeshing run without a valid position attribute");
167 }
168 passes.emplace_back(
169 Pass{
170 m_meshes.get_mesh_path(position_attr.mesh()),
171 1,
172 {"split", "collapse", "swap", "smooth"}});
173 } else {
174 passes = opts.passes;
175 }
176 for (size_t j = 0; j < passes.size(); ++j) {
177 auto& pass = passes[j];
178 if (pass.operations.empty()) {
179 wmtk::logger().info(
180 "Pass of isotropic remeshing didn't specify any operations, "
181 "assuming it's a split,collapse,swap,smooth");
182 pass.operations = {"split", "collapse", "swap", "smooth"};
183 }
184 for (const auto& op_name : pass.operations) {
185 if (m_operations.find(op_name) == m_operations.end()) {
186 throw std::runtime_error(
187 fmt::format("was unable to find operation {} in pass {}", op_name, j));
188 }
189 }
190 }
191 for (const auto& [name, op] : m_operations) {
192 if (!op->attribute_new_all_configured()) {
193 wmtk::log_and_throw_error("Not every attribute in {} was configured", name);
194 }
195 }
196}

222void IsotropicRemeshing::make_interior_invariants(const IsotropicRemeshingOptions& opts)
223{
224 auto position = get_attribute(opts.position_attribute);
225 Mesh& mesh = position.mesh();
226 auto invariant_interior_vertex = std::make_shared<wmtk::invariants::InvariantCollection>(mesh);
227 m_interior_edge_invariants = std::make_shared<wmtk::invariants::InvariantCollection>(mesh);
228
229 auto set_all_invariants = [&](auto&& m) {
230 // TODO: this used to do vertex+edge, but just checkign for vertex should be sufficient?
231 for (PrimitiveType pt = PrimitiveType::Vertex; pt < m.top_simplex_type(); pt = pt + 1) {
232 invariant_interior_vertex->add(
233 std::make_shared<wmtk::invariants::InteriorSimplexInvariant>(m, pt));
234 }
235
236 m_interior_edge_invariants->add(
237 std::make_shared<wmtk::invariants::InteriorSimplexInvariant>(m, PrimitiveType::Edge));
238 };
239 wmtk::multimesh::MultiMeshVisitor visitor(set_all_invariants);
240 visitor.execute_from_root(mesh);
241 m_interior_position_invariants = invariant_interior_vertex;
242}

244void IsotropicRemeshing::run()
245{
246 wmtk::logger().info(
247 "Running {} different types of passes for {} iterations",
248 passes.size(),
249 iterations);
250
251 if (m_split) {
252 wmtk::logger().debug(
253 "Isotropic Remeshing split has invariant [{}]",
254 m_split->invariants().name());
255 }
256 if (m_collapse) {
257 wmtk::logger().debug(
258 "Isotropic Remeshing collapse has invariant [{}]",
259 m_collapse->invariants().name());
260 }
261 if (m_swap) {
262 wmtk::logger().debug(
263 "Isotropic Remeshing swap has invariant [{}], and its split has [{}], collapse has "
264 "[{}]",
265 m_swap->invariants().name(),
266 m_swap->split().invariants().name(),
267 m_swap->collapse().invariants().name());
268 }
269 if (m_smooth) {
270 wmtk::logger().debug(
271 "Isotropic Remeshing smooth has invariant [{}]",
272 m_smooth->invariants().name());
273 }
274
275 auto log_mesh = [&](int64_t index) {
276 for (const auto& [name, mesh] : m_meshes.all_meshes()) {
277 spdlog::info(
278 "Mesh [{}] has {} {}-facets",
279 name,
280 mesh.get_all(mesh.top_simplex_type()).size(),
281 mesh.top_cell_dimension());
282 }
283 if (intermediate_output_format.empty()) {
284 wmtk::logger().error(
285 "Failed to log using intermediate_output_format because "
286 "no MeshCollection was attached");
287 return;
288 }
289 for (const auto& [name, opts] : intermediate_output_format) {
290 auto opt = wmtk::components::output::utils::format(opts, index);
291 spdlog::info("Temp logging {} as {}", name, opt.path.string());
292 wmtk::components::output::output(m_meshes.get_mesh(name), opt);
293 }
294 };
295
296 log_mesh(0);
297 size_t index = 0;
298 for (size_t k = 1; k <= iterations; ++k) {
299 for (size_t j = 0; j < passes.size(); ++j) {
300 Pass& p = passes[j];
301 wmtk::logger().info(
302 "Running pass {}/{} of iteration {}/{}. Has {} sub-iterations on {})",
303 j,
304 passes.size(),
305 k,
306 iterations,
307 p.iterations,
308 p.mesh_path);
309 run(p, j);
310 }
311 log_mesh(k);
312 }

400void IsotropicRemeshing::make_envelopes(const IsotropicRemeshingOptions& opts)
401{
402 if (!opts.envelope_size.has_value()) {
403 throw std::runtime_error(
404 "Could not create envelopes because options do not have an envelope size");
405 }
406 auto envelope_positions = all_envelope_positions(opts);
407
408 std::vector<std::shared_ptr<wmtk::invariants::EnvelopeInvariant>> envelope_invariants;
409
410
411 std::transform(
412 envelope_positions.begin(),
413 envelope_positions.end(),
414 std::back_inserter(envelope_invariants),
415 [&](const wmtk::attribute::MeshAttributeHandle& mah) {
416 spdlog::warn(
417 "Envelope made for {}",
418 primitive_type_name(mah.mesh().top_simplex_type()));
419 return std::make_shared<wmtk::invariants::EnvelopeInvariant>(
420 mah,
421 std::sqrt(2) * opts.envelope_size.value(),
422 mah);
423 });
424
425 m_envelope_invariants = std::make_shared<wmtk::invariants::InvariantCollection>(
426 get_attribute(opts.position_attribute).mesh().get_multi_mesh_root());
427
428 for (const auto& invar : envelope_invariants) {
429 m_envelope_invariants->add(invar);
430 }
431}
