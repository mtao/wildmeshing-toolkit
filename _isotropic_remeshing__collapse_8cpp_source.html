56void IsotropicRemeshing::configure_collapse(const IsotropicRemeshingOptions& opts)
57{
58 wmtk::logger().debug("Configure isotropic remeshing collapse");
59 wmtk::Mesh& mesh = get_attribute(opts.position_attribute).mesh();
60 auto& op = m_collapse = std::make_shared<operations::EdgeCollapse>(mesh);
61
62
63 add_core_collapse_invariants(*op, opts);
64 if (opts.separate_substructures) {
65 auto invariant_separate_substructures =
66 std::make_shared<wmtk::invariants::SeparateSubstructuresInvariant>(mesh);
67 op->add_invariant(invariant_separate_substructures);
68 }
69
70
71 const double length_min = (4. / 5.) * opts.get_absolute_length(m_meshes);
72 auto pos_attr = get_attribute(opts.position_attribute);
73
74 auto tmp = std::make_shared<operations::CollapseNewAttributeStrategy<double>>(pos_attr);
75 tmp->set_strategy(operations::CollapseBasicStrategy::Mean);
76 tmp->set_simplex_predicate(operations::BasicSimplexPredicate::IsInterior);
77 op->set_new_attribute_strategy(pos_attr, tmp);
78
79 auto invariant_max_edge_length = std::make_shared<MaxEdgeLengthInvariant>(
80 pos_attr.mesh(),
81 pos_attr.as<double>(),
82 length_min * length_min);
83 op->add_invariant(invariant_max_edge_length);
84
85 if (!opts.static_meshes.empty()) {
86 for (const auto& mesh_name : opts.static_meshes) {
87 auto& mesh2 = m_meshes.get_mesh(mesh_name);
88 op->add_invariant(
89 std::make_shared<invariants::CannotMapSimplexInvariant>(
90 mesh,
91 mesh2,
92 wmtk::PrimitiveType::Vertex));
93 }
94 }
95
96 if (m_envelope_invariants) {
97 spdlog::info("Attaching envelope invariants");
98 op->add_invariant(m_envelope_invariants);
99 }
100
101 for (const auto& [c, p] : opts.copied_attributes) {
102 auto parent = get_attribute(p);
103 auto child = get_attribute(c);
104 op->set_new_attribute_strategy(child, operations::CollapseBasicStrategy::None);
105
106 op->add_transfer_strategy(
107 wmtk::operations::attribute_update::make_cast_attribute_transfer_strategy(
108 parent,
109 child));
110 }
111 for (const auto& transfer : m_operation_transfers) {
112 spdlog::info("Adding collapse transfer for {}", transfer->handle().name());
113 op->set_new_attribute_strategy(transfer->handle(), operations::CollapseBasicStrategy::None);
114 op->add_transfer_strategy(transfer);
115 }
116 for (const auto& attr : opts.pass_through_attributes) {
117 op->set_new_attribute_strategy(get_attribute(attr));
118 }
119
120 if (opts.collapse.priority) {
121 opts.collapse.priority->assign_to(m_meshes, *op);
122 }
123
124 if (m_universal_invariants) {
125 op->add_invariant(m_universal_invariants);
126 }
127
128 assert(op->attribute_new_all_configured());
129}
