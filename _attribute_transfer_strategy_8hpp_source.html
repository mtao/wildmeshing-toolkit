176template <typename MyType, typename ParentType, int MyDim, int ParentDim>
177auto SingleAttributeTransferStrategy<MyType, ParentType, MyDim, ParentDim>::read_parent_values(
178 const simplex::Simplex& my_simplex) const -> std::pair<ParentMatType, std::vector<Tuple>>
179{
180 auto acc =
181 parent_handle().mesh().create_const_accessor(parent_handle().template as<ParentType>());
182 auto simps =
183 AttributeTransferStrategyBase::get_parent_simplices(handle(), parent_handle(), my_simplex);
184
185//#if !defined(NDEBUG)
186 if (simps.size() == 0) {
187// throw std::runtime_error(fmt::format("SingleAttributeTransferStrategy got no simplices {}/{} to {}/{}",fmt::join(parent_handle().mesh().absolute_multi_mesh_id(),","), parent_handle().name(),fmt::join(handle().mesh().absolute_multi_mesh_id(),","), handle().name() ));
188 return {};
189 }
190//#endif
191
192 ParentMatType A(
193 parent_handle().mesh().get_attribute_dimension(parent_handle().template as<ParentType>()),
194 simps.size());
195
196 using Index = Eigen::Index;
197 for (Index j = 0; j < Index(simps.size()); ++j) {
198 A.col(j) = acc.template const_vector_attribute<ParentDim>(simps[j]);
199 }
200 return std::make_pair(std::move(A), std::move(simps));
201}
202template <typename MyType, typename ParentType, int MyDim, int ParentDim>
203void SingleAttributeTransferStrategy<MyType, ParentType, MyDim, ParentDim>::run(
204 const simplex::Simplex& s) const
205{
206 assert(mesh().is_valid(s.tuple()));
207 if (s.primitive_type() != primitive_type()) {
208 // TODO: is this an error out or silent fail
209 return;
210 }
211
212 if (m_functor) {
213 auto [parent_data, simps] = read_parent_values(s);
214 if (simps.empty()) return;
215 auto acc = const_cast<Mesh&>(mesh()).create_accessor(handle().template as<MyType>());
216
217 acc.template vector_attribute<MyDim>(s.tuple()) = m_functor(parent_data, simps);
218 }
