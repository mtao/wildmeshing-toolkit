140inline int64_t TetMesh::id(const Tuple& tuple, PrimitiveType type) const
141{
142 switch (type) {
143 case PrimitiveType::Vertex: {
144 auto tv = m_tv_accessor->const_vector_attribute<4>(tuple);
145 return tv(tuple.local_vid());
146 break;
147 }
148 case PrimitiveType::Edge: {
149 auto te = m_te_accessor->const_vector_attribute<6>(tuple);
150 return te(tuple.local_eid());
151 break;
152 }
153 case PrimitiveType::Triangle: {
154 auto tf = m_tf_accessor->const_vector_attribute<4>(tuple);
155 return tf(tuple.local_fid());
156 break;
157 }
158 case PrimitiveType::Tetrahedron: {
159 return tuple.global_cid();
160 break;
161 }
162 default: assert(false); // "Tuple id: Invalid primitive type"
163 }
164
165 return -1;
166}

168inline int64_t TetMesh::id(int64_t global_id, int8_t permutation_index, PrimitiveType pt) const
169{
170 int8_t index = autogen::tet_mesh::simplex_index_from_permutation_index(permutation_index, pt);
171 switch (pt) {
172 case PrimitiveType::Vertex: {
173 return m_tv_accessor->attribute().const_vector_single_value(global_id, index);
174 }
175 case PrimitiveType::Edge: {
176 return m_te_accessor->attribute().const_vector_single_value(global_id, index);
177 }
178 case PrimitiveType::Triangle: {
179 return m_tf_accessor->attribute().const_vector_single_value(global_id, index);
180 }
181 case PrimitiveType::Tetrahedron: {
182 return global_id;
183 break;
184 }
185 default: assert(false); // "Tuple id: Invalid primitive type"
186 }
