21auto compute_tet_amips = [](const Eigen::MatrixXd& P) -> Eigen::VectorXd {
22 // tet
23 std::array<double, 12> pts;
24 for (size_t i = 0; i < 4; ++i) {
25 for (size_t j = 0; j < 3; ++j) {
26 pts[3 * i + j] = P(j, i);
27 }
28 }
29 const double a = wmtk::function::utils::Tet_AMIPS_energy(pts);
30 return Eigen::VectorXd::Constant(1, a);
31};

164 auto swap44_energy_check = [&](int64_t idx, const simplex::Simplex& t) -> double {
165 constexpr static PrimitiveType PV = PrimitiveType::Vertex;
166 constexpr static PrimitiveType PE = PrimitiveType::Edge;
167 constexpr static PrimitiveType PF = PrimitiveType::Triangle;
168 constexpr static PrimitiveType PT = PrimitiveType::Tetrahedron;
169
170 assert(options.position_attribute.holds<wmtk::Rational>());
171 auto accessor = mesh.create_const_accessor(options.position_attribute.as<Rational>());
172
173 // get the coords of the vertices
174 // input edge end points
175 const Tuple e0 = t.tuple();
176 const Tuple e1 = mesh.switch_tuple(e0, PV);
177 // other four vertices
178 std::array<Tuple, 4> v;
179 auto iter_tuple = e0;
180 for (int64_t i = 0; i < 4; ++i) {
181 v[i] = mesh.switch_tuples(iter_tuple, {PE, PV});
182 iter_tuple = mesh.switch_tuples(iter_tuple, {PF, PT});
183 }
184
185 if (iter_tuple != e0) return 0;
186 assert(iter_tuple == e0);
187
188 std::array<Eigen::Vector3<Rational>, 6> positions = {
189 {accessor.const_vector_attribute(v[(idx + 0) % 4]),
190 accessor.const_vector_attribute(v[(idx + 1) % 4]),
191 accessor.const_vector_attribute(v[(idx + 2) % 4]),
192 accessor.const_vector_attribute(v[(idx + 3) % 4]),
193 accessor.const_vector_attribute(e0),
194 accessor.const_vector_attribute(e1)}};
195 std::array<Eigen::Vector3d, 6> positions_double = {
196 {positions[0].cast<double>(),
197 positions[1].cast<double>(),
198 positions[2].cast<double>(),
199 positions[3].cast<double>(),
200 positions[4].cast<double>(),
201 positions[5].cast<double>()}};
202
203 std::array<std::array<int, 4>, 4> new_tets = {
204 {{{0, 1, 2, 4}}, {{0, 2, 3, 4}}, {{0, 1, 2, 5}}, {{0, 2, 3, 5}}}};
205
206 double new_energy_max = std::numeric_limits<double>::lowest();
207
208 for (int i = 0; i < 4; ++i) {
209 if (wmtk::utils::wmtk_orient3d(
210 positions[new_tets[i][0]],
211 positions[new_tets[i][1]],
212 positions[new_tets[i][2]],
213 positions[new_tets[i][3]]) > 0) {
214 auto energy = wmtk::function::utils::Tet_AMIPS_energy({{
215 positions_double[new_tets[i][0]][0],
216 positions_double[new_tets[i][0]][1],
217 positions_double[new_tets[i][0]][2],
218 positions_double[new_tets[i][1]][0],
219 positions_double[new_tets[i][1]][1],
220 positions_double[new_tets[i][1]][2],
221 positions_double[new_tets[i][2]][0],
222 positions_double[new_tets[i][2]][1],
223 positions_double[new_tets[i][2]][2],
224 positions_double[new_tets[i][3]][0],
225 positions_double[new_tets[i][3]][1],
226 positions_double[new_tets[i][3]][2],
227 }});
228
229 if (energy > new_energy_max) new_energy_max = energy;
230 } else {
231 auto energy = wmtk::function::utils::Tet_AMIPS_energy({{
232 positions_double[new_tets[i][1]][0],
233 positions_double[new_tets[i][1]][1],
234 positions_double[new_tets[i][1]][2],
235 positions_double[new_tets[i][0]][0],
236 positions_double[new_tets[i][0]][1],
237 positions_double[new_tets[i][0]][2],
238 positions_double[new_tets[i][2]][0],
239 positions_double[new_tets[i][2]][1],
240 positions_double[new_tets[i][2]][2],
241 positions_double[new_tets[i][3]][0],
242 positions_double[new_tets[i][3]][1],
243 positions_double[new_tets[i][3]][2],
244 }});
245
246 if (energy > new_energy_max) new_energy_max = energy;
247 }
248 }
249
250 return new_energy_max;
251 };

310std::shared_ptr<wmtk::operations::Operation> tet_swap(
311 TetMesh& mesh,
312 const IsotropicRemeshingOptions& options)
313{
314 auto swap_all = std::make_shared<wmtk::operations::OrOperationSequence>(mesh);
315 //swap_all->add_operation(tet_swap32(mesh,options));
316 swap_all->add_operation(tet_swap44(mesh,options));
317 swap_all->add_operation(tet_swap56(mesh,options));
318
319
321 // energy filter flag
323 auto& position_mesh = const_cast<Mesh&>(options.position_attribute.mesh());
324 auto energy_filter_handle =
325 position_mesh
326 .register_attribute<char>("energy_filter", PrimitiveType::Vertex, 1, false, char(1));
327
328 auto energy_filter_accessor = position_mesh.create_accessor<char>(energy_filter_handle);
329
330 auto update_energy_filter_func = [](const Eigen::MatrixX<double>& P) -> Eigen::VectorX<char> {
331 return Eigen::VectorX<char>::Constant(1, char(1));
332 };
333 auto energy_filter_update =
334 std::make_shared<wmtk::operations::SingleAttributeTransferStrategy<char, double>>(
335 energy_filter_handle,
336 options.position_attribute,
337 update_energy_filter_func);
338
340 // renew flags
342 auto visited_edge_flag_handle =
343 position_mesh
344 .register_attribute<char>("visited_edge", PrimitiveType::Edge, 1, false, char(1));
345
346 auto update_flag_func = [](const Eigen::MatrixX<double>& P) -> Eigen::VectorX<char> {
347 return Eigen::VectorX<char>::Constant(1, char(1));
348 };
349 auto tag_update =
350 std::make_shared<wmtk::operations::SingleAttributeTransferStrategy<char, double>>(
351 visited_edge_flag_handle,
352 options.position_attribute,
353 update_flag_func);
354
355 swap_all->add_transfer_strategy(tag_update);
356 swap_all->add_transfer_strategy(energy_filter_update);
357 return swap_all;
358}
