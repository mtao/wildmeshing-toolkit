43int8_t edge_mirror_left_action(const SimplexDart& sd, int8_t edge_permutation_index)
44{
45 const int8_t SV = sd.primitive_as_index(PrimitiveType::Vertex);
46
47 int8_t edge_permutation_index_inv = sd.inverse(edge_permutation_index);
48 int8_t tmp = sd.product(edge_permutation_index, SV);
49 int8_t r = sd.product(tmp, edge_permutation_index_inv);
