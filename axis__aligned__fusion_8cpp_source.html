32std::shared_ptr<Mesh> axis_aligned_fusion(
33 const attribute::MeshAttributeHandle& position_handle,
34 const std::vector<bool>& operating_axis,
35 double eps)
36{
37 Mesh& mesh = const_cast<Mesh&>(position_handle.mesh());
38 // get mesh dimension and checks
39 int64_t mesh_dim = mesh.top_cell_dimension();
40
41 if (mesh_dim != position_handle.dimension()) {
42 throw std::runtime_error(fmt::format(
43 "Can only perform axis aligned fusion on a {0}-dimensional position on a {0}-mesh, got "
44 "a {1}-dimensional position",
45 mesh_dim,
46 position_handle.dimension()));
47 }
48 if (mesh_dim != operating_axis.size()) {
49 throw std::runtime_error(fmt::format(
50 "Can only perform axis aligned fusion on a {0} entries in the operating_axis mask on a "
51 "{0}-mesh, got a {1} values",
52 mesh_dim,
53 operating_axis.size()));
54 }
55
56 std::map<std::string, std::vector<int64_t>> names;
57 wmtk::utils::EigenMatrixWriter writer;
58 MatrixX<double> V;
59 mesh.serialize(writer);
60 writer.get_position_matrix(V);
61
62
63 // rescale to [0, 1]
64 Eigen::MatrixXd V_rescale(V.rows(), V.cols());
65
66 Eigen::VectorXd min_pos = V.colwise().minCoeff();
67
68 // if (abs(min_pos[2] - 0.0) > eps)
69 // throw std::runtime_error("has non-zero z coordinate on 2d mesh");
70
71 for (int64_t i = 0; i < V_rescale.rows(); ++i) {
72 V_rescale.row(i) = V.row(i) - min_pos.transpose();
73 }
74
75 Eigen::VectorXd max_pos = V_rescale.colwise().maxCoeff();
76 V_rescale.array().rowwise() /= max_pos.transpose().array();
77 assert(V.rows() == mesh.get_all(PrimitiveType::Vertex).size());
78
79
80 int dim = std::min<int>(operating_axis.size(), V.cols());
81 auto comp = [&V_rescale, &operating_axis, dim, eps](int ai, int bi) {
82 auto a = V_rescale.row(ai);
83 auto b = V_rescale.row(bi);
84
85 auto d = (b - a).cwiseAbs();
86
87 for (size_t j = 0; j < dim; ++j) {
88 if (operating_axis[j]) {
89 // d < 1, so can skip the abs
90 if (d(j) > eps && d(j) < 1 - eps) {
91 return false;
92 }
93 } else {
94 if (d(j) > eps) {
95 return false;
96 }
97 }
98 }
99 return true;
100 };
101 std::vector<std::set<int64_t>> fusable_min(dim);
102 std::vector<std::set<int64_t>> fusable_max(dim);
103 for (int j = 0; j < V.rows(); ++j) {
104 auto v = V.row(j);
105 for (int d = 0; d < dim; ++d) {
106 if (operating_axis[d]) {
107 if (v(d) <= eps) {
108 fusable_min[d].emplace(j);
109 } else if (v(d) + eps >= 1.0) {
110 fusable_max[d].emplace(j);
111 }
112 }
113 }
114 }
115 wmtk::utils::DisjointSet ds(V.rows());
116 for (int d = 0; d < dim; ++d) {
117 for (const auto& m : fusable_min[d]) {
118 for (const auto& M : fusable_max[d]) {
119 const bool res = comp(m, M);
120 if (res) {
121 ds.merge(m, M);
122 }
123 }
124 }
125 }
126
127 std::map<int64_t, int64_t> root_indexer;
128 auto roots = ds.roots();
129 for (size_t j = 0; j < roots.size(); ++j) {
130 root_indexer[roots[j]] = j;
131 }
132
133
134 auto get_index = [&ds, &root_indexer](int64_t i) { return root_indexer.at(ds.get_root(i)); };
135
136
137 std::shared_ptr<Mesh> parent_ptr;
138 // TODO: a lot of matrix copies to remove
139
140 switch (mesh_dim) {
141 case (2): {
142 MatrixX<int64_t> FV;
143
144 // TODO: make this the mah passed in
145 writer.get_FV_matrix(FV);
146
147 assert(FV.rows() == mesh.get_all(PrimitiveType::Triangle).size());
148
149 // create periodic mesh
150 RowVectors3l FV_new = FV.unaryExpr(get_index);
151
152 std::shared_ptr<TriMesh> parent = std::make_shared<TriMesh>();
153 parent->initialize(FV_new);
154 parent_ptr = parent;
155 break;
156 }
157 case (3): {
158 MatrixX<int64_t> TV;
159
160 writer.get_TV_matrix(TV);
161
162 assert(TV.rows() == mesh.get_all(PrimitiveType::Tetrahedron).size());
163
164 // // create periodic mesh
165 RowVectors4l TV_new = TV.unaryExpr(get_index);
166 std::shared_ptr<TetMesh> parent = std::make_shared<TetMesh>();
167 parent->initialize(TV_new);
168 parent_ptr = parent;
169
170
171 break;
172 }
173 default: {
174 throw std::runtime_error("mesh dimension not supported");
175 }
176 }
177 auto child_map = wmtk::multimesh::same_simplex_dimension_bijection(*parent_ptr, mesh);
178 parent_ptr->register_child_mesh(mesh.shared_from_this(), child_map);
179 return parent_ptr;
180}

201void MultimeshAxisAlignedFusionOptions::run(MeshCollection& mc) const
202{
203 auto attr = utils::get_attribute(mc, attribute_path);
204 assert(attr.mesh().is_multi_mesh_root());
205 auto [mesh_path, attr_name] = utils::decompose_attribute_path(attribute_path.path);
206 const auto& child_nmm = mc.get_named_multimesh(mesh_path);

208 assert(!attr.mesh().is_multi_mesh_root());

212 nmm.append_child_mesh_names(*mesh, child_nmm);
