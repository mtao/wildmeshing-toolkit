30// compute the length relative to the bounding box diagonal
31double relative_to_absolute_length(
32 const attribute::MeshAttributeHandle& position,
33 const double length_rel)
34{
35 if (!position.is_valid()) {
36 throw std::runtime_error(
37 "Could not convert relative to absolute length because position attr was invalid");
38 }
39 auto pos = position.mesh().create_const_accessor<double>(position);
40 const auto vertices = position.mesh().get_all(PrimitiveType::Vertex);
41 Eigen::AlignedBox<double, Eigen::Dynamic> bbox(pos.dimension());
42
43
44 for (const auto& v : vertices) {
45 bbox.extend(pos.const_vector_attribute(v));
46 }
47
48 const double diag_length = bbox.sizes().norm();
49 wmtk::logger().debug(
50 "computed absolute target length using relative factor {} on bbox diagonal {}",
51 length_rel,
52 diag_length);
53
54 return length_rel * diag_length;
55}

58double IsotropicRemeshingOptions::get_absolute_length() const
59{
60 double length = length_abs;
61 if (length_abs <= 0) {
62 if (length_rel <= 0) {
63 throw std::runtime_error("Either absolute or relative length must be set!");
64 }
65 length = relative_to_absolute_length(position_attribute, length_rel);
66 } else {
67 wmtk::logger().debug(
68 "get_absolute_length using absolute length value {} {}",
69 length,
70 length_abs);
71 }
72 return length;
