31// compute the length relative to the bounding box diagonal
32double relative_to_absolute_length(
33 const attribute::MeshAttributeHandle& position,
34 const double length_rel)
35{
36 if (!position.is_valid()) {
37 throw std::runtime_error(
38 "Could not convert relative to absolute length because position attr was invalid");
39 }
40 auto pos = position.mesh().create_const_accessor<double>(position);
41 const auto vertices = position.mesh().get_all(PrimitiveType::Vertex);
42 Eigen::AlignedBox<double, Eigen::Dynamic> bbox(pos.dimension());
43
44
45 for (const auto& v : vertices) {
46 bbox.extend(pos.const_vector_attribute(v));
47 }
48
49 const double diag_length = bbox.sizes().norm();
50 wmtk::logger().debug(
51 "computed absolute target length using relative factor {} on bbox diagonal {}",
52 length_rel,
53 diag_length);
54
55 return length_rel * diag_length;
56}

59double IsotropicRemeshingOptions::get_absolute_length(const multimesh::MeshCollection& mc) const
60{
61 double length = length_abs;
62 if (length_abs <= 0) {
63 if (length_rel <= 0) {
64 throw std::runtime_error("Either absolute or relative length must be set!");
65 }
66 auto position_attr = multimesh::utils::get_attribute(mc, position_attribute);
67 length = relative_to_absolute_length(position_attr, length_rel);
68 } else {
69 wmtk::logger().debug(
70 "get_absolute_length using absolute length value {} {}",
71 length,
72 length_abs);
73 }
74 return length;
75}
