35template <size_t D, size_t E>
36std::vector<std::array<int64_t, E>> get_simplices(std::array<int64_t, D> s)
37{
38 static_assert(E <= D);
39 std::set<std::array<int64_t, E>> F;
40
41 std::array<int64_t, E> r;
42
43 // extract the first E elements of each permutation of s
44 std::sort(s.begin(), s.end());
45 do {
46 std::copy(s.begin(), s.begin() + E, r.begin());
47 std::sort(r.begin(), r.end());
48 // append to the simplex set
49 F.emplace(r);
50 } while (std::next_permutation(s.begin(), s.end()));
51 return std::vector<std::array<int64_t, E>>{F.begin(), F.end()};
52}
