211bool MapValidator::check_child_facet_uniqueness(const Mesh& child) const
212{
213 bool ok = true;
214 assert(child.m_multi_mesh_manager.m_parent == &m_mesh);
215 const auto pt = child.top_simplex_type();
216
217 auto child_facets = child.get_all(pt);
218
219
220 std::vector<std::vector<simplex::Simplex>> children_mapped(child_facets.size());
221
222
223 for (const auto& parent_t : m_mesh.get_all(pt)) {
224 wmtk::simplex::Simplex s(pt, parent_t);
225 auto child_ss = m_mesh.map_to_child(child, s);
226 for (const auto& child_s : child_ss) {
227 auto child_id = child.id(child_s);
228 if(child_id >= children_mapped.size()) {
229 children_mapped.resize(child_id + 1);
230 }
231 children_mapped.at(child_id).emplace_back(child_s);
232
233 }
234 }
235 for (size_t j = 0; j < children_mapped.size(); ++j) {
236 const auto& children = children_mapped.at(j);
237 if (children.size() > 1) {
238 std::vector<std::string> tups;
239 ok = false;
240 std::transform(
241 children.begin(),
242 children.end(),
243 std::back_inserter(tups),
244 [](const auto& x) { return std::string(x.tuple()); });
245 wmtk::logger().error(
246 "Single child facet maps to multiple parent facets: [{}]:{} maps to [{}]:{}",
247 fmt::join(m_mesh.absolute_multi_mesh_id(), ","),
248 j,
249 fmt::join(child.absolute_multi_mesh_id(), ","),
250 fmt::join(tups, ","));
251 }
252 }
253 return ok;

256bool MapValidator::check_child_switch_homomorphism(const Mesh& child) const
257{
258 assert(child.m_multi_mesh_manager.m_parent == &m_mesh);
259 bool ok = true;
260 for (PrimitiveType pt : wmtk::utils::primitive_below(child.top_simplex_type())) {
261 auto tups = child.get_all(pt);
262 for (const wmtk::Tuple& t : tups) {
263 assert(!t.is_null());
264 wmtk::simplex::Simplex s(pt, t);
265
266 wmtk::Tuple parent_tuple = child.map_to_parent_tuple(s);
267 assert(!parent_tuple.is_null());
268 assert(m_mesh.is_valid(parent_tuple));
269
270 // spt is the primitive type we will switch
271 for (PrimitiveType spt : wmtk::utils::primitive_below(child.top_simplex_type())) {
272 if (pt == PrimitiveType::Vertex) {
273 continue;
274 }
275 // skip switches over boundaries
276 if (spt == child.top_simplex_type()) {
277 continue;
278 }
279 if (spt == child.top_simplex_type() &&
280 child.is_boundary(child.top_simplex_type() - 1, t)) {
281 continue;
282 }
283
284 wmtk::simplex::Simplex switched_simplex(pt, child.switch_tuple(t, spt));
285 wmtk::Tuple switch_map = child.map_to_parent_tuple(switched_simplex);
286 wmtk::Tuple map_switch = m_mesh.switch_tuple(parent_tuple, spt);
287 for (PrimitiveType my_pt : wmtk::utils::primitive_below(child.top_simplex_type())) {
288 bool worked = wmtk::simplex::utils::SimplexComparisons::equal(
289 m_mesh,
290 my_pt,
291 switch_map,
292 map_switch);
293 if (!worked) {
294 wmtk::logger().error(
295 "Map from child {0} to parent {1} on tuple {2} (dim {3}) fails on "
296 "switch(map({2}),{4}) = {5} != "
297 "map(switch({2},{4})) = {6} (dim {7} id {8} != {9})",
298 fmt::join(child.absolute_multi_mesh_id(), ","), // 0
299 fmt::join(m_mesh.absolute_multi_mesh_id(), ","), // 1
300 t.as_string(), // 2
301 child.top_cell_dimension(), // 3
302 primitive_type_name(pt), // 4
303 map_switch.as_string(), // 5
304 switch_map.as_string(), // 6
305 primitive_type_name(my_pt), // 7
306 m_mesh.id(map_switch, my_pt), // 8
307 m_mesh.id(switch_map, my_pt) // 9
308 );
309 ok = false;
310 }
311 }
312 }
313 }
314 }
315 return ok;
316}
