182bool MapValidator::check_child_facet_uniqueness(const Mesh& child) const
183{
184 bool ok = true;
185 assert(child.m_multi_mesh_manager.m_parent == &m_mesh);
186 const auto pt = child.top_simplex_type();
187
188 auto child_facets = child.get_all(pt);
189
190
191 std::vector<std::vector<simplex::Simplex>> children_mapped(child_facets.size());
192
193
194 for (const auto& parent_t : m_mesh.get_all(pt)) {
195 wmtk::simplex::Simplex s(pt, parent_t);
196 auto child_ss = m_mesh.map_to_child(child, s);
197 for (const auto& child_s : child_ss) {
198 auto child_id = child.id(child_s);
199 if (child_id >= children_mapped.size()) {
200 children_mapped.resize(child_id + 1);
201 }
202 children_mapped.at(child_id).emplace_back(child_s);
203 }
204 }
205 for (size_t j = 0; j < children_mapped.size(); ++j) {
206 const auto& children = children_mapped.at(j);
207 if (children.size() > 1) {
208 std::vector<std::string> tups;
209 ok = false;
210 std::transform(
211 children.begin(),
212 children.end(),
213 std::back_inserter(tups),
214 [](const auto& x) { return std::string(x.tuple()); });
215 wmtk::logger().error(
216 "Single child facet maps to multiple parent facets: [{}]:{} maps to [{}]:{}",
217 fmt::join(m_mesh.absolute_multi_mesh_id(), ","),
218 j,
219 fmt::join(child.absolute_multi_mesh_id(), ","),
220 fmt::join(tups, ","));
221 }
222 }
223 return ok;

226bool MapValidator::check_child_switch_homomorphism(const Mesh& child) const
227{
228 assert(child.m_multi_mesh_manager.m_parent == &m_mesh);
229 bool ok = true;
230 for (PrimitiveType pt : wmtk::utils::primitive_below(child.top_simplex_type())) {
231 auto tups = child.get_all(pt);
232 for (const wmtk::Tuple& t : tups) {
233 assert(!t.is_null());
234 wmtk::simplex::Simplex s(pt, t);
235
236 wmtk::Tuple parent_tuple = child.map_to_parent_tuple(s);
237 assert(!parent_tuple.is_null());
238 assert(m_mesh.is_valid(parent_tuple));
239
240 // spt is the primitive type we will switch
241 for (PrimitiveType spt : wmtk::utils::primitive_below(child.top_simplex_type())) {
242 if (pt == PrimitiveType::Vertex) {
243 continue;
244 }
245 // skip switches over boundaries
246 if (spt == child.top_simplex_type()) {
247 continue;
248 }
249 if (spt == child.top_simplex_type() &&
250 child.is_boundary(child.top_simplex_type() - 1, t)) {
251 continue;
252 }
253
254 wmtk::simplex::Simplex switched_simplex(pt, child.switch_tuple(t, spt));
255 wmtk::Tuple switch_map = child.map_to_parent_tuple(switched_simplex);
256 wmtk::Tuple map_switch = m_mesh.switch_tuple(parent_tuple, spt);
257 for (PrimitiveType my_pt : wmtk::utils::primitive_below(child.top_simplex_type())) {
258 bool worked = wmtk::simplex::utils::SimplexComparisons::equal(
259 m_mesh,
260 my_pt,
261 switch_map,
262 map_switch);
263 if (!worked) {
264 wmtk::logger().error(
265 "Map from child [{0}] to parent [{1}] on tuple {2} (dim {3}) fails on "
266 "switch(map({2}),{4}) = {5} != "
267 "map(switch({2},{4})) = {6} (dim {7} id {8} != {9})",
268 fmt::join(child.absolute_multi_mesh_id(), ","), // 0
269 fmt::join(m_mesh.absolute_multi_mesh_id(), ","), // 1
270 t.as_string(), // 2
271 child.top_cell_dimension(), // 3
272 primitive_type_name(pt), // 4
273 map_switch.as_string(), // 5
274 switch_map.as_string(), // 6
275 primitive_type_name(my_pt), // 7
276 m_mesh.id(map_switch, my_pt), // 8
277 m_mesh.id(switch_map, my_pt) // 9
278 );
279 ok = false;
280 }
281 }
282 }
283 }
284 }
285 return ok;
