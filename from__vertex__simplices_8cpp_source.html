56template <size_t Dim, typename MeshType>
57std::vector<std::array<Tuple, 2>> from_vertex_simplices_map_(
58 MeshType& parent,
59 const wmtk::utils::internal::IndexSimplexMapper& indexer,
60 const MatrixXl& l)
61{
62 using VecType = wmtk::Vector<int64_t, Dim>;
63 using MapType = VecType::MapType;
64
65 const auto& sd = wmtk::dart::SimplexDart::get_singleton(parent.top_simplex_type());
66 std::vector<std::array<Tuple, 2>> R;
67 R.reserve(l.rows());
68 for (int j = 0; j < l.rows(); ++j) {
69 std::array<int64_t, Dim> r;
70 MapType(r.data()) = l.row(j);
71 R.emplace_back(std::array<Tuple, 2>{
72 {sd.tuple_from_dart(wmtk::dart::Dart(j, sd.identity())), indexer.get_tuple(r)}});
73 }
74 return R;
75}

102std::shared_ptr<Mesh> from_vertex_simplices(Mesh& parent, const MatrixXl& l)
103{
104 auto child_map = from_vertex_simplices_map(parent, l);
105
106 MatrixXl l2;
107 std::map<int64_t, int64_t> indexer;
108 for (const auto& j : std::span(l.data(), l.data() + l.size())) {
109 int64_t a = indexer.size();
110 indexer.try_emplace(j, a);
111 }
112 l2 = l.unaryExpr([&](int64_t a) { return indexer.at(a); });
113 const auto pt = get_primitive_type_from_id(l.cols() - 1);
114
115 std::shared_ptr<Mesh> child;
116 switch (pt) {
117 default:
118 case PrimitiveType::Vertex: break;
119 case PrimitiveType::Edge: {
120 auto m = std::make_shared<EdgeMesh>();
121 m->initialize(l2);
122 return m;
123 }
124 case PrimitiveType::Triangle: {
125 auto m = std::make_shared<TriMesh>();
126 m->initialize(l2);
127 return m;
128 }
129 case PrimitiveType::Tetrahedron: {
130 auto m = std::make_shared<TetMesh>();
131 m->initialize(l2);
132 return m;
133 }
134 }
135
136 parent.register_child_mesh(child, child_map);
137
138
139 return child;
140}
