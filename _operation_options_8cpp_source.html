24void PriorityOptions::assign_to(
25 const wmtk::components::multimesh::MeshCollection& mc,
26 wmtk::operations::Operation& op) const
27{
28 if (type == "random") {
29 op.use_random_priority() = false;
30 op.set_priority(nullptr);
31 } else if (type == "random") {
32 op.use_random_priority() = true;
33 op.set_priority(nullptr);
34 } else if (type == "attribute") {
35 op.use_random_priority() = false;
36 auto ap = attribute_path;
37 if (!ap.empty()) {
38 auto priority_attribute = wmtk::components::multimesh::utils::get_attribute(
39 mc,
40 wmtk::components::multimesh::utils::AttributeDescription{ap});
41 if (minimize) {
42 auto priority_func = [priority_attribute](const simplex::Simplex& s) -> double {
43 auto acc =
44 priority_attribute.mesh().create_const_accessor<double>(priority_attribute);
45 return acc.const_scalar_attribute(s);
46 };
47 op.set_priority(priority_func);
48 } else {
49 auto priority_func = [priority_attribute](const simplex::Simplex& s) -> double {
50 auto acc =
51 priority_attribute.mesh().create_const_accessor<double>(priority_attribute);
52 return -acc.const_scalar_attribute(s);
53 };
54 op.set_priority(priority_func);
55 }
56 }
57 } else {
58 assert(false); // tried to assign an invalid priority
59 }
60}

227 } else {

230 "Expected edge_swap_mode to be one of [amips,valence,skip], got [{}]",
