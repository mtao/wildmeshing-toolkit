197RowVectors<int64_t, Dim> manifold_decomposition(Eigen::Ref<const RowVectors<int64_t, Dim>> S)

206template <size_t Dim>
207wmtk::RowVectors<int64_t, Dim - 1> ManifoldDecomposition<Dim>::face_matrix(
208 bool manifold_topology,
209 bool compactify) const
210{
211 size_t total_size = 0;
212 if (manifold_topology) {
213 for (const auto& [k, v] : face_map) {
214 total_size += v.size();
215 }
216 } else {
217 total_size = face_map.size();
218 }
219 wmtk::RowVectors<int64_t, Dim - 1> R(total_size, Dim - 1);
220 Eigen::Index i = 0;
221 const PrimitiveType pt = get_primitive_type_from_id(Dim - 1);
222 for (const auto& [key, values] : face_map) {
223 using MT = typename wmtk::Vector<int64_t, Dim - 1>::ConstMapType;
224
225 if (manifold_topology) {
226 for (const dart::Dart& v : values) {
227 auto s = manifold_decomposition.row(v.global_id());
228 auto row = dart::utils::permute(s, v.permutation());
229 R.row(i++) = row.head(Dim - 1).transpose();
230 }
231 } else {
232 R.row(i++) = MT(key.data()).transpose();
233 }
234 }
235 if (compactify) {
236 R = compactify_eigen_indices(R);
237 }
238 return R;
239}

241template <size_t Dim>
242std::vector<std::array<Tuple, 2>> ManifoldDecomposition<Dim>::mm_map(bool manifold_topology) const
243{
244 size_t total_size = 0;
245 if (manifold_topology) {
246 for (const auto& [k, v] : face_map) {
247 total_size += v.size();
248 }
249 } else {
250 total_size = face_map.size();
251 }
252 std::vector<std::array<Tuple, 2>> tups;
253 tups.reserve(total_size);
254 Eigen::Index i = 0;
255 const auto& parent_sd = dart::SimplexDart::get_singleton(get_primitive_type_from_id(Dim - 1));
256 const auto& sd = dart::SimplexDart::get_singleton(get_primitive_type_from_id(Dim - 2));
257 int8_t id = sd.identity();
258 for (const auto& [key, values] : face_map) {
259 for (const auto& value : values) {
260 Tuple t = sd.tuple_from_dart(wmtk::dart::Dart(i++, id));
261 Tuple parent_t = parent_sd.tuple_from_dart(value);
262 // tups.emplace_back(std::array<Tuple, 2>{{parent_t, t}});
263 tups.emplace_back(std::array<Tuple, 2>{{t, parent_t}});
264 if (!manifold_topology) { // in manifold topology i only need 1
265 break;
266 }
267 }
268 }
269 return tups;
