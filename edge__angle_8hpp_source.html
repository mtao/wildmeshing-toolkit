11// given k-1 simplex normals, produce a k-2 facet
12template <typename InT, int InDim, typename OutT, int OutDim>
13struct EdgeAngleFunctor
14{
15 EdgeAngleFunctor(const nlohmann::json& js) {}
16 constexpr static bool validInDim()
17 {
18 return InDim == Eigen::Dynamic || InDim == 2 || InDim == 3;
19 }
20 constexpr static bool validOutDim() { return OutDim == Eigen::Dynamic || OutDim == 1; }
21 constexpr static bool validType()
22 {
23 return std::is_same_v<double, InT> && std::is_same_v<double, OutT>;
24 }
25 // takes as input normals
26 constexpr static bool valid() { return validInDim() && validOutDim() && validType(); }
27
28 static auto execute(Eigen::Ref<const ColVectors<InT, InDim>> M) -> Vector<OutT, OutDim>
29 {
30 if (M.cols() == 1) {
31 return Vector<OutT, 1>::Constant(OutT(0));
32 }
33 assert(M.cols() == 2);
34
35 auto p0 = M.col(0);
36 auto p1 = M.col(1);
37 if constexpr (InDim == 2) {
38 auto cos = p1.dot(p0);
39
40 auto sin = std::sqrt(p0.homogeneous().cross(p1.homogeneous()).z());
41
42 return Vector<OutT, 1>::Constant(std::atan2(sin, cos));
43
44 } else if constexpr (InDim == 3) {
45 auto cos = p1.dot(p0);
46 auto sin = p0.cross(p1).norm();
47 return Vector<OutT, 1>::Constant(std::atan2(sin, cos));
48
49 } else if constexpr (InDim == Eigen::Dynamic) {
50 // run as static sizes
51 if (M.rows() == 2) {
52 return execute(M.template topRows<2>());
53 } else if (M.rows() == 3) {
54 return execute(M.template topRows<3>());
55 }
56 }
57 assert(false);
58
59 return {};
60 }
61 auto operator()(Eigen::Ref<const ColVectors<InT, InDim>> M) const -> Vector<OutT, OutDim>
62 {
63 return execute(M);
64 }
65};
