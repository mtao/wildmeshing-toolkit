13template <typename T>
14std::pair<MatrixXl, std::vector<Tuple>> extract_child_simplices_and_map_from_tag(
15 const Mesh& m,
16 const wmtk::attribute::TypedAttributeHandle<T>& tag_handle,
17 const T& tag_value,
18 bool decompose_nonmanifold)
19{
20 PrimitiveType pt = tag_handle.primitive_type();
21 std::vector<Tuple> tagged_tuples = utils::tagged_tuples<T>(m, tag_handle, tag_value);
22 if (tagged_tuples.empty()) {
23 return {};
24 }
25 MatrixXl S;
26 wmtk::utils::EigenMatrixWriter em;
27 m.serialize(em);
28 MatrixXl F = em.get_simplex_vertex_matrix();
29 const dart::SimplexDart& mesh_sd = dart::SimplexDart::get_singleton(m.top_simplex_type());
30 S.resize(tagged_tuples.size(), get_primitive_type_id(pt) + 1);
31
32 for (size_t j = 0; j < tagged_tuples.size(); ++j) {
33 const dart::Dart d = mesh_sd.dart_from_tuple(tagged_tuples[j]);
34 auto f = F.row(d.global_id());
35 auto perm = dart::utils::permute(f, d.permutation());
36 S.row(j) = perm.transpose().head(S.cols());
37 }
38
39 S = wmtk::utils::internal::compactify_eigen_indices(S);
40 if (decompose_nonmanifold) {
41 switch (S.cols()) {
42 case 1: S = wmtk::utils::internal::manifold_decomposition<1>(S); break;
43 case 2: S = wmtk::utils::internal::manifold_decomposition<2>(S); break;
44 case 3: S = wmtk::utils::internal::manifold_decomposition<3>(S); break;
45 case 4: S = wmtk::utils::internal::manifold_decomposition<4>(S); break;
46 default: break;
47 }
48 }
49
50 return std::make_pair(S, tagged_tuples);
51}

86template <typename T>
87std::vector<Tuple> tagged_tuples(
88 const Mesh& m,
89 const wmtk::attribute::TypedAttributeHandle<T>& tag_handle,
90 const T& tag_value)
91{
92 auto tags = m.create_const_accessor(tag_handle);
93 const PrimitiveType pt = tag_handle.primitive_type();
94
95 std::vector<Tuple> tagged_tuples;
96 for (const Tuple& t : m.get_all(pt)) {
97 if (tags.const_scalar_attribute(t) == tag_value) {
98 tagged_tuples.emplace_back(t);
99 }
100 }
101 return tagged_tuples;
102}
